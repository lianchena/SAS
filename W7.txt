LIBNAME ACX "";
OPTION ERROR=3;

/*1.winsorize data*/
*construct variables and delete missing values;
DATA ROE; SET ACX.ROE;
ROE=NI/CEQ;
PM=NI/SALE;
ATO=SALE/AT;
LEV=AT/CEQ;

ARRAY M(*) ROE PM ATO LEV;
DO I=1 TO HBOUD(M);
IF M(i)=. THEN DELETE;
END;
KEEP GVKEY DATADATE FYEAR SICI ROE PM ATO LEV;
RUN;

PROC MEANS DATA=ROW;
VAR ROE PM ATO LEV;
RUN;
